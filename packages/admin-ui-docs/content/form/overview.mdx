### Validation

```jsx noInline live
// import { Form, TextInput, NumberInput, useFormState } from '@vtex/admin-ui-form'

const { Form, TextInput, NumberInput, useFormState } = Forms

function Example() {
  const form = useFormState()

  const handleSubmit = (data) => {
    alert(JSON.stringify(data))
  }

  return (
    <Form state={form} onSubmit={handleSubmit}>
      <Stack space="$2xl">
        <TextInput
          label="Name"
          name="name"
          helpText="Enter your name"
          validation={{
            required: { value: true, message: 'This is a required field' },
          }}
          state={form}
        />

        <TextInput
          label="Email"
          name="Email"
          helpText="Enter your email"
          validation={{
            pattern: {
              value: RegExp(/^[a-z0-9.]+@[a-z0-9]+\.[a-z]+(\.[a-z]+)?$/i),
              message: 'Email is not valid',
            },
          }}
          state={form}
        />

        <NumberInput
          label="Age"
          name="age"
          helpText="Enter your age"
          validation={{
            min: { value: 18, message: 'The minimum age is 18' },
            required: { value: true, message: 'This is a required field' },
          }}
          state={form}
        />

        <Button type="submit">Submit</Button>
      </Stack>
    </Form>
  )
}

render(<Example />)
```

### Validation using Yup

Yup is a schema builder for runtime value parsing and validation

```jsx noInline live
// import { Form, TextInput, TextArea, yupResolver, useFormState } from '@vtex/admin-ui-form'

const { Form, TextInput, TextArea, yupResolver, useFormState } = Forms

function Example() {
  const schema = Yup.object({
    firstName: Yup.string().required(),
    comment: Yup.string().required().min(10, 'Make a longer comment'),
  })

  const form = useFormState({
    resolver: yupResolver(schema),
  })

  const handleSubmit = (data) => {
    alert(JSON.stringify(data))
  }

  return (
    <Form state={form} onSubmit={handleSubmit}>
      <Stack space="$2xl">
        <TextInput label="First Name" name="firstName" state={form} />
        <TextArea label="Comment" state={form} name="comment" />

        <Button type="submit">Submit</Button>
      </Stack>
    </Form>
  )
}

render(<Example />)
```

### useWatch

```jsx noInline live
// import { Form, useWatch, useFormState } from '@vtex/admin-ui-form'

const { Form, useWatch, useFormState } = Forms

function Example() {
  const form = useFormState()

  const handleSubmit = (data) => {
    alert(JSON.stringify(data))
  }

  const firstName = useWatch({
    form,
    name: 'firstName',
  })

  const age = useWatch({
    form,
    name: 'age',
  })

  return (
    <Form state={form} onSubmit={handleSubmit}>
      <Stack space="$2xl">
        <Box
          as="input"
          csx={{ border: '$neutral' }}
          placeholder="First Name"
          {...form.register('firstName')}
        />
        <Box
          as="input"
          csx={{ border: '$neutral' }}
          placeholder="Age"
          type="number"
          {...form.register('age')}
        />
        <Button type="submit">Submit</Button>
        <Box>
          <Box as="p">firstName: {firstName}</Box>
          <Box as="p">age: {age}</Box>
        </Box>
      </Stack>
    </Form>
  )
}

render(<Example />)
```
