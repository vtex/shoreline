## Usage

```jsx
import {
  Card,
  CardHeader,
  CardInfo,
  CardTitle,
  CardActions,
  CardContent,
  CardImage,
} from '@vtex/admin-ui'

export function Example() {
  return (
    <Card>
      <CardHeader>
        <CardInfo>
          <IconImageSquare />
          <CardTitle>Title</CardTitle>
          <Tag label="Short text" />
        </CardInfo>
        <CardActions>
          <Button variant="tertiary">Label</Button>
          <Button variant="secondary">Label</Button>
        </CardActions>
      </CardHeader>
      <CardContent>Content</CardContent>
    </Card>
  )
}
```
