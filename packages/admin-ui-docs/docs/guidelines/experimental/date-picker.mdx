## Usage

```jsx
import {
  experimental_DatePickerField as DatePickerField,
  experimental_DatePickerCalendar as DatePickerCalendar,
  experimental_useDatePickerState as useDatePickerState,
} from '@vtex/admin-ui'

function Example() {
  const state = useDatePickerState()
  return <DatePicker label="Date" state={state} />
}
```

## Examples

### Default value

```jsx noInline live
const DatePickerField = experimental_DatePickerField
const DatePickerCalendar = experimental_DatePickerCalendar
const useDatePickerState = experimental_useDatePickerState

function Example() {
  const state = useDatePickerState({
    defaultValue: {
      year: 2020,
      month: 5,
      day: 20,
    },
  })

  return (
    <>
      <DatePickerField label="Date" state={state} />
      <DatePickerCalendar state={state} />
    </>
  )
}

render(<Example />)
```

### Min and Max values

```jsx noInline live
const DatePickerField = experimental_DatePickerField
const DatePickerCalendar = experimental_DatePickerCalendar
const useDatePickerState = experimental_useDatePickerState

function Example() {
  const state = useDatePickerState({
    minValue: {
      year: 2022,
      month: 0,
      day: 20,
    },
    maxValue: {
      year: 2024,
      month: 11,
      day: 20,
    },
  })

  return (
    <>
      <DatePickerField
        label="Date"
        state={state}
        criticalText="Date out of bounds"
      />
      <DatePickerCalendar state={state} />
    </>
  )
}

render(<Example />)
```

### Validation

Use the `invalid` boolean to trigger the invalid state. It is up to you to define what is valid or not. Additionally, you can add a `criticalText` to be displayed whether invalid.

```jsx noInline live
const DatePickerField = experimental_DatePickerField
const DatePickerCalendar = experimental_DatePickerCalendar
const useDatePickerState = experimental_useDatePickerState

function Example() {
  const state = useDatePickerState({
    invalid: true,
  })

  return (
    <>
      <DatePickerField
        label="Date"
        state={state}
        criticalText="This date is invalid"
      />
      <DatePickerCalendar state={state} />
    </>
  )
}

render(<Example />)
```

### Format

You can set formmating of date based on [Intl.DateFormatOptions](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Intl/DateTimeFormat) using the `formatOptions` property.

```jsx noInline live
const DatePickerField = experimental_DatePickerField
const DatePickerCalendar = experimental_DatePickerCalendar
const useDatePickerState = experimental_useDatePickerState

function Example() {
  const state = useDatePickerState({
    formatOptions: {
      year: 'numeric',
      month: 'short',
      day: '2-digit',
      weekday: 'long',
    },
  })

  return (
    <>
      <DatePickerField label="Date" state={state} />
      <DatePickerCalendar state={state} />
    </>
  )
}

render(<Example />)
```

### I18n

The date-picker has default translations. Wrap your app with the [I18nProvider](/experimental/localization#i18nprovider/) to configure the locale.

```jsx noInline live
const DatePickerField = experimental_DatePickerField
const DatePickerCalendar = experimental_DatePickerCalendar
const useDatePickerState = experimental_useDatePickerState
const I18nProvider = experimental_I18nProvider

function DatePickerComponent() {
  const state = useDatePickerState()
  return (
    <>
      <DatePickerField label="Date" state={state} />
      <DatePickerCalendar state={state} />
    </>
  )
}

function RootApp() {
  return (
    <I18nProvider locale="ko-KO">
      <DatePickerComponent />
    </I18nProvider>
  )
}

render(<RootApp />)
```

### Placeholder

Each segment placeholder is defined individually. Placeholders don't have default translation.

```jsx noInline live
const DatePickerField = experimental_DatePickerField
const DatePickerCalendar = experimental_DatePickerCalendar
const useDatePickerState = experimental_useDatePickerState

function Example() {
  const state = useDatePickerState({
    placeholder: {
      day: 'DD',
      month: 'MM',
      year: 'YYYY',
    },
  })

  return (
    <>
      <DatePickerField label="Date" state={state} />
      <DatePickerCalendar state={state} />
    </>
  )
}

render(<Example />)
```
