# Admin UI Label

Form label component

## Usage

- Simple usage

```jsx
import { Label } from '@vtex/admin-ui'

function UseCase() {
  return (
    <Label>
      Input Label!
      <input value="input value" />
    </Label>
  )
}
```

- Using `htmlFor` prop

```jsx
import { Label } from '@vtex/admin-ui'

function UseCase() {
  return (
    <>
      <Label htmlFor="my-input">Input Label!</Label>
      <input id="my-input" value="input value" />
    </>
  )
}
```
